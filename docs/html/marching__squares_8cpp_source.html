49double grad(int hash, double x, double y, double z)
50{
51 int h = hash & 15;
52 double u = h < 8 ? x : y,
53 v = h < 4 ? y : h == 12 || h == 14 ? x
54 : z;
55 return ((h & 1) == 0 ? u : -u) + ((h & 2) == 0 ? v : -v);
56}

58void noise(double x, double y, double z, double& value)
59{
60 int X = (int)floor(x) & 255,
61 Y = (int)floor(y) & 255,
62 Z = (int)floor(z) & 255;
63 x -= floor(x);
64 y -= floor(y);
65 z -= floor(z);
66 double u = fade(x),
67 v = fade(y),
68 w = fade(z);
69 int A = p[X] + Y, AA = p[A] + Z, AB = p[A + 1] + Z,
70 B = p[X + 1] + Y, BA = p[B] + Z, BB = p[B + 1] + Z;
71
72 double val = lerp(w, lerp(v, lerp(u, grad(p[AA], x, y, z), grad(p[BA], x - 1, y, z)), lerp(u, grad(p[AB], x, y - 1, z), grad(p[BB], x - 1, y - 1, z))),
73 lerp(v, lerp(u, grad(p[AA + 1], x, y, z - 1), grad(p[BA + 1], x - 1, y, z - 1)),
74 lerp(u, grad(p[AB + 1], x, y - 1, z - 1),
75 grad(p[BB + 1], x - 1, y - 1, z - 1))));
76 value = val * 0.5 + 0.5;
77}

79void GetNoiseVec(int x, int y,double time, Vector<Vector<double>>& vec)
80{
81 for (int i = -1000 / g_Dim; i <= 1000 / g_Dim; i++)
82 {
83 Vector<double> temp;
84 temp.reserve(1000 / g_Dim * 2 + 1);
85 for (int j = -1000 / g_Dim; j <= 1000 / g_Dim; j++)
86 {
87 double v;
88 noise((i + x) * g_DimPerlin, (j + y) * g_DimPerlin, time, std::ref(v));
89 temp.push_back(std::move(v));
90 }
91 vec.emplace_back(std::move(temp));
92 }
93}

96{
